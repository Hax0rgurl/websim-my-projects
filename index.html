    🤖 Bot/Developer Guide: Fetching Projects

    If you are a bot or a developer trying to fetch a user's projects programmatically, follow this pattern. The API uses cursor-based pagination.

    The Algorithm

    
async function fetchAllUserProjects(username) {
    let allProjects = [];
    let hasNextPage = true;
    let afterCursor = null;

    while (hasNextPage) {
        // 1. Construct URL parameters
        const params = new URLSearchParams();
        params.append('first', '100'); // Maximize page size
        if (afterCursor) {
            params.append('after', afterCursor);
        }

        // 2. Fetch the data
        const response = await fetch(`/api/v1/users/${username}/projects?` + params.toString());
        const data = await response.json();

        // 3. Process the page
        if (data?.projects?.data) {
            // 'data.projects.data' contains the array of project items
            allProjects = allProjects.concat(data.projects.data);
            
            // 4. Update cursor for next iteration
            hasNextPage = data.projects.meta.has_next_page;
            afterCursor = data.projects.meta.end_cursor;
        } else {
            hasNextPage = false;
        }
    }
    
    return allProjects;
}

    Data Structure Note

    Each item in the returned array contains:

    • project: { id, title, updated_at, stats: { views, likes }, visibility }
    • site: { id } (Used for constructing thumbnail URLs: https://images.websim.ai/v1/site/{site_id}/600)
